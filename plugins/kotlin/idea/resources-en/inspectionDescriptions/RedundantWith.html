Reports a redundant with function call that doesn't access anything from the receiver.

Examples:


class MyClass {
    fun f(): String = ""
}

fun testOk() {
    val c = MyClass()
    with(c) { // <== it's ok since 'f()' is effectively 'c.f()'
        println(f())
    }
}

fun testRedundant() {
    with (c) { // <== 'with' is redundant since 'c' isn't used
        println("1")
    }
}
